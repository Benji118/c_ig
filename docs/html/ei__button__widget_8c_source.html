98 static ei_linked_point_t* ei_arc(ei_point_t centre, float R, float angledebut, float anglefin)
99 {
100 
101  ei_linked_point_t* liste = NULL;
102 
103  float angle = angledebut;
104 
105  while (angle < anglefin)
106  {
107  ei_linked_point_t* new_point = malloc(sizeof(ei_linked_point_t));
108  new_point->point = ei_point( (int)( R * cos(angle)) + centre.x, (int)(R * sin(angle)) + centre.y );
109  new_point->next = NULL;
110  concat_lists(&liste, &new_point);
111  // Pas de rotation = 0.1
112  angle = angle + 0.1;
113  }
114 
115  ei_linked_point_t* last_point = malloc(sizeof(ei_linked_point_t));
116  last_point->point = ei_point( (int)( R * cos(anglefin)) + centre.x, (int)(R * sin(anglefin)) + centre.y );
117  last_point->next = NULL;
118 
119  concat_lists(&liste, &last_point);
120 
121  return liste;
122 }

309 void ei_draw_relief_button(struct ei_widget_t* widget,
310  ei_surface_t surface,
311  ei_surface_t pick_surface,
312  ei_rect_t* clipper)
313 {
314  ei_button_widget_t* button_widget = (ei_button_widget_t*) widget;
315 
316  ei_color_t gray = { 0x33, 0x33, 0x33, 0xAA };
317  ei_color_t white = { 0xD0, 0xD0, 0xD0, 0xFF };
318  ei_color_t black = { 0x30, 0x30, 0x30, 0xFF };
319 
320  ei_color_t relief_color = ei_clearer_color(button_widget->color);
321 
322  ei_rect_t rectangle = widget->screen_location;
323 
324  int border = button_widget->border_width;
325 
326  ei_linked_point_t* high_list = ei_curvy_contour_button( EI_TRUE, EI_FALSE,
327  rectangle, button_widget->corner_radius, 0);
328 
329  ei_linked_point_t* low_list = ei_curvy_contour_button( EI_FALSE, EI_TRUE,
330  rectangle, button_widget->corner_radius, 0);
331 
332  ei_linked_point_t* all_list = ei_curvy_contour_button( EI_TRUE, EI_TRUE,
333  rectangle, button_widget->corner_radius, border);
334 
335 
336 
337  switch ( button_widget->relief ) {
338  case ei_relief_none :
339  break;
340  case ei_relief_raised: // Couleur claire en haut et couleur foncée en bas
341  ei_draw_polygon(surface, high_list, white, clipper);
342  ei_draw_polygon(surface, high_list, relief_color, clipper);
343  ei_draw_polygon(surface, low_list, black, clipper);
344  ei_draw_polygon(surface, low_list, relief_color, clipper);
345  ei_draw_polyline(surface, low_list, gray, clipper);
346  ei_draw_polyline(surface, high_list, gray, clipper);
347  break;
348  case ei_relief_sunken: // Couleur foncée en haut et couleur claire en bas
349  ei_draw_polygon(surface, high_list, black, clipper);
350  ei_draw_polygon(surface, high_list, relief_color, clipper);
351  ei_draw_polygon(surface, low_list, white, clipper);
352  ei_draw_polygon(surface, low_list, relief_color, clipper);
353  ei_draw_polyline(surface, low_list, gray, clipper);
354  ei_draw_polyline(surface, high_list, gray, clipper);
355  break;
356  }
357 
358  ei_draw_polygon(surface, all_list, button_widget->color, clipper);
359  ei_color_t pick_pick = *(widget->pick_color);
360  ei_draw_polygon(pick_surface, all_list, pick_pick, clipper);
361  ei_draw_polyline(surface, all_list, gray, clipper);
362 
363  ei_free_linked_points_list(high_list);
364  ei_free_linked_points_list(low_list);
365  ei_free_linked_points_list(all_list);
366 
367 }
