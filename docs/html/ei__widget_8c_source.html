274 ei_widget_t* ei_widget_pick(ei_point_t* where)
275 {
276  // Pointeur sur le pixel en haut à gauche (origine) de l'offscreen
277  uint8_t* pointeur = hw_surface_get_buffer(offscreen_surface);
278 
279  // Taille de l'offscreen

281 
282  // Déplacement de "pointeur" dans la mémoire :
283  // "pointeur" pointe sur le pixel correspondant à "where"
284  pointeur = pointeur + ( (where->y)*offscreen_size.width + where->x ) * 4;
285 
286  // On réccupère l'ordre dans la mémoire des couleurs d'un pixel
287  int red;
288  int green;
289  int blue;
290  int alpha;
291  hw_surface_get_channel_indices(offscreen_surface,&red,&green,&blue,&alpha);
292 
293  // Puis on récupère chaque couleur du pixel pointé par "pointeur"
294  uint8_t* pointeur_red = pointeur + red;
295  uint8_t* pointeur_green = pointeur + green;
296  uint8_t* pointeur_blue = pointeur + blue;
297 
298  // Calcul de l'entier correspondant à la couleur du pixel
299  uint32_t n = (*pointeur_blue) + 256 * (*pointeur_green) + 256 * 256 * (*pointeur_red);
300 
301  // Maintenant on retrouve le widget correspondant à l'entier
302  ei_widget_t* root_widget = ei_app_root_widget();
303  ei_widget_t* solution;
304 
305  find_widget_from_pick_id(root_widget, n, &solution);
306 
307  return(solution);
