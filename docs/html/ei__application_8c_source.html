94 // Insertion d'un rectangle en queue de la liste globale
95 static void insertion_tail(ei_linked_rect_t* new_rect)
96 {
97  // 1er cas : liste vide
98  if (rects_to_update == NULL) {
99  rects_to_update = new_rect;
100  // 2nd cas : liste contenant au moins un élément
101  } else {
102  ei_linked_rect_t* courant = rects_to_update;
103  // On se place en derniere position
104  while (courant->next != NULL) {
105  courant = courant->next;
106  }
107  // On chaîne le nouvel élément
108  courant->next = new_rect;
109  }
110 }

206 // Fonction récursive qui dessine le widget donné en paramètre ainsi que
207 // toute sa descendance (parcours en profondeur)
208 static void ei_draw_widget_family(ei_widget_t* widget)
209 {
210  // Vérification du widget passé en paramètre
211  if (widget == NULL) {
212  return;
213  }
214 
215  // Classe du widget à dessiner
216  ei_widgetclass_t* widget_class = widget->wclass;
217 
218  // Définition du paramètre "clipper"
219  // @param clipper : If not NULL, the drawing is restricted within this rectangle
220  // (expressed in the surface reference frame).
221  ei_rect_t clipper;
222  if ( widget->parent == NULL ) {
223  clipper = hw_surface_get_rect(display_surface);
224  } else if ( (widget->parent)->content_rect == NULL ) {
225  clipper = hw_surface_get_rect(display_surface);
226  } else {
227  clipper = *(widget->parent)->content_rect;
228  }
229 
230  // Dessin du widget
231  widget_class->drawfunc(widget,display_surface,offscreen_surface,&clipper);
232 
233  // Appels récursifs sur les fils du widget
234  ei_widget_t* fils_courant = widget->children_head;
235  while (fils_courant != NULL) {
236  ei_draw_widget_family(fils_courant);
237  fils_courant = fils_courant->next_sibling;
238  }
239 }

267 // Fonction qui exécute les callbacks lors d'un événement
268 void ei_execute_callbacks(ei_widget_t* widget, ei_event_t* event)
269 {
270  ei_bindingcell_t* handle = bindinglist;
271  ei_rect_t clipped_rect;
272 
273  // mise à jour de la mémoire d'enfoncement de la souris
274  if (event->type == ei_ev_mouse_buttondown) {
275  set_mouse_down(EI_TRUE);
276  set_mouse_down_coordinates(event->param.mouse.where);
277  set_current_id(widget->pick_id);
278  }
279  if (event->type == ei_ev_mouse_buttonup) {
280  set_mouse_down(EI_FALSE);
281  }
282 
283  // on parcourt toute la bindinglist
284  while (handle != NULL) {
285  // On repère les bindings correspondant aux widgets concernés ou au tag "all"
286  if ( (handle->binding.widget = widget)
287  || (!strncmp(handle->binding.tag,"all",3))
288  || (!strncmp(handle->binding.tag,widget->wclass->name,20))) {
289  // On regarde si c'est le bon event
290  if (handle->binding.eventtype == event->type) {
291  // Si c'est le cas, on exécute le callback avec les paramètres spécifiés
292  handle->binding.callback(widget, event, handle->binding.user_param);
293  // on invalidate les rects en les empêchant de dépasser du root_widget
294  clipped_rect = ei_rectangles_intersect(*widget->content_rect, ei_app_root_widget()->screen_location);
295  ei_app_invalidate_rect(&clipped_rect);
296  }
297  }
298  handle = handle->next;
299  }
300 }

308 // appelle les drawfunc de tous les widgets
309 // en les parcourant en profondeur

324  clipper = handle_rect->rect;
